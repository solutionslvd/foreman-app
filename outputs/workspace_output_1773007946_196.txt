            "files_saved": 0,
            "auto_replies_sent": 0,
            "leads_converted": 0
        }
    }

    users_db[email] = user
    logger.info(f"New user registered: {email} ({business_name})")
    return user


def authenticate_user(email: str, password: str) -> Optional[Dict]:
    """Authenticate a user"""
    user = users_db.get(email)
    if not user:
        return None
    if not verify_password(password, user["password_hash"]):
        return None
